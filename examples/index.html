Default Example No configuration

  1. Click on the <> icon.
  2. Choose your langage and type some code.
  3. Click on the Source button to see the output of the plugin.
  4. Check the Result panel to see the live result with a Syntax Highlighter (or not, there is no highlighter loaded for this example).
CKEDITOR.editorConfig = function( config ) {
    config.toolbarGroups = [
        // shows the source button
        { name: 'document', groups: [ 'mode'] },
        // shows the pbckcode button
        { name: 'pbckcode' }
    ];
    config.extraPlugins = 'pbckcode';
};

Google Code Prettify

CKEDITOR.editorConfig = function( config ) {
    config.toolbarGroups = [
        // shows the source button
        { name: 'document', groups: [ 'mode'] },
        // shows the pbckcode button
        { name: 'pbckcode' }
    ];
    config.extraPlugins = 'pbckcode';
    config.pbckcode = {
        highlighter : 'PRETTIFY'
        modes :  [ ['C', 'c_pp'] ],
        theme : 'clouds',
    };
}
